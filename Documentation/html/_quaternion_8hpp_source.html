550  {
551  TQuaternion<T> R;
552 
553  // Calculate angle between them.
554  T CosHalfTheta = a.DotProduct(b);
555 
556  // if qa=qb or qa=-qb then theta = 0 and we can return qa
557  if (Abs(CosHalfTheta) >= T(1))
558  {
559  R.w = a.w;
560  R.x = a.x;
561  R.y = a.y;
562  R.z = a.z;
563  return R;
564  }
565 
566  // Calculate temporary values.
567  T HalfTheta = Acos(CosHalfTheta);
568  T SinHalfTheta = Sqrt(T(1) - CosHalfTheta * CosHalfTheta);
569 
570  // if theta = 180 degrees then result is not fully defined
571  // we could rotate around any axis normal to qa or qb
572  if (Abs(SinHalfTheta) < T(0.001))
573  { // fabs is floating point absolute
574  R.w = (a.w * T(0.5) + b.w * T(0.5));
575  R.x = (a.x * T(0.5) + b.x * T(0.5));
576  R.y = (a.y * T(0.5) + b.y * T(0.5));
577  R.z = (a.z * T(0.5) + b.z * T(0.5));
578  return R;
579  }
580 
581  T RatioA = Sin((T(1) - v) * HalfTheta) / SinHalfTheta;
582  T RatioB = Sin(v * HalfTheta) / SinHalfTheta;
583 
584  //calculate Quaternion.
585  R.w = (a.w * RatioA + b.w * RatioB);
586  R.x = (a.x * RatioA + b.x * RatioB);
587  R.y = (a.y * RatioA + b.y * RatioB);
